        document.addEventListener('DOMContentLoaded', () => {
            const grid = document.getElementById('feature-grid');
            featureData.forEach((feature, index) => {
                const button = document.createElement('div');
                button.className = 'tech-button rounded-3xl p-6 flex flex-col items-center text-center';
                button.setAttribute('data-index', index);
                button.onclick = () => showModal(index);

                button.innerHTML = `
                    <div class="text-6xl mb-3">${feature.icon}</div>
                    <h3 class="text-xl font-extrabold text-white">${feature.title}</h3>
                    ${feature.creditCost ? `<span class="mt-2 text-sm font-semibold px-3 py-1 bg-cyan-800/50 text-cyan-300 rounded-full">${feature.creditCost}</span>` : ''}
                `;
                grid.appendChild(button);
            });
        });

        const modal = document.getElementById('feature-modal');
        const modalTitle = document.getElementById('modal-title');
        const modalContentBody = document.getElementById('modal-content-body');

        function showModal(index) {
            const feature = featureData[index];
            if (!feature) {
                console.error("Feature not found for index:", index);
                return;
            }

            modalTitle.innerHTML = feature.title;
            modalContentBody.innerHTML = feature.content;
            modal.classList.remove('hidden');
            document.body.classList.add('overflow-hidden'); // Prevent background scrolling
        }

        function closeModal(event) {
            // Check if the click occurred on the backdrop (modal itself) or the close button
            if (!event || event.currentTarget === modal || event.target.tagName === 'BUTTON' || event.target.classList.contains('text-3xl')) {
                modal.classList.add('hidden');
                document.body.classList.remove('overflow-hidden');
            }
        }
